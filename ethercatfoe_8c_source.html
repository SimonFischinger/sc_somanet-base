94 int ec_FOEread(uint16 slave, char *filename, uint32 password, int *psize, void *p, int timeout)
95 {
96  ec_FOEt *FOEp, *aFOEp;
97  int wkc;
98  int32 dataread = 0;
99  int32 buffersize, packetnumber, prevpacket = 0;
100  uint16 fnsize, maxdata, segmentdata;
101  ec_mbxbuft MbxIn, MbxOut;
102  uint8 cnt;
103  boolean worktodo;
104 
105  buffersize = *psize;
106  ec_clearmbx(&MbxIn);
107  /* Empty slave out mailbox if something is in. Timout set to 0 */
108  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, 0);
109  ec_clearmbx(&MbxOut);
110  aFOEp = (ec_FOEt *)&MbxIn;
111  FOEp = (ec_FOEt *)&MbxOut;
112  fnsize = strlen(filename);
113  maxdata = ec_slave[slave].mbx_l - 12;
114  if (fnsize > maxdata)
115  fnsize = maxdata;
116  printf("\n max data %d\n",maxdata);
117 maxdata=512;
118  FOEp->MbxHeader.length = htoes(0x0006 + fnsize);
119  FOEp->MbxHeader.address = htoes(0x0000);
120  FOEp->MbxHeader.priority = 0x00;
121  /* get new mailbox count value, used as session handle */
122  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
123  ec_slave[slave].mbx_cnt = cnt;
124  FOEp->MbxHeader.mbxtype = ECT_MBXT_FOE + (cnt << 4); /* FoE */
125  FOEp->OpCode = ECT_FOE_READ;
126  FOEp->Password = htoel(password);
127  /* copy filename in mailbox */
128  memcpy(&FOEp->FileName[0], filename, fnsize);
129  /* send FoE request to slave */
130  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
131  if (wkc > 0) /* succeeded to place mailbox in slave ? */
132  {
133  do
134  {
135  worktodo = FALSE;
136  /* clean mailboxbuffer */
137  ec_clearmbx(&MbxIn);
138  /* read slave response */
139  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, timeout);
140  if (wkc > 0) /* succeeded to read slave response ? */
141  {
142  /* slave response should be FoE */
143  if ((aFOEp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_FOE)
144  {
145  if(aFOEp->OpCode == ECT_FOE_DATA)
146  {
147  segmentdata = etohs(aFOEp->MbxHeader.length) - 0x0006;
148  packetnumber = etohl(aFOEp->PacketNumber);
149  if ((packetnumber == ++prevpacket) && (dataread + segmentdata <= buffersize))
150  {
151  memcpy(p, &aFOEp->Data[0], segmentdata);
152  dataread += segmentdata;
153  p += segmentdata;
154  if (segmentdata == maxdata)
155  worktodo = TRUE;
156  FOEp->MbxHeader.length = htoes(0x0006);
157  FOEp->MbxHeader.address = htoes(0x0000);
158  FOEp->MbxHeader.priority = 0x00;
159  /* get new mailbox count value */
160  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
161  ec_slave[slave].mbx_cnt = cnt;
162  FOEp->MbxHeader.mbxtype = ECT_MBXT_FOE + (cnt << 4); /* FoE */
163  FOEp->OpCode = ECT_FOE_ACK;
164  FOEp->PacketNumber = htoel(packetnumber);
165  /* send FoE ack to slave */
166  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
167  if (wkc <= 0)
168  worktodo = FALSE;
169  }
170  else
171  {
172  /* FoE error */
173  wkc = -EC_ERR_TYPE_FOE_BUF2SMALL;
174  }
175  }
176  else
177  if(aFOEp->OpCode == ECT_FOE_ERROR)
178  {
179  /* FoE error */
180  wkc = -EC_ERR_TYPE_FOE_ERROR;
181  }
182  else
183  {
184  /* unexpected mailbox received */
185  wkc = -EC_ERR_TYPE_PACKET_ERROR;
186  }
187  }
188  else
189  {
190  /* unexpected mailbox received */
191  wkc = -EC_ERR_TYPE_PACKET_ERROR;
192  }
193  *psize = dataread;
194  }
195  } while (worktodo);
196  }
197 
198  return wkc;
199 }
