68 void ec_dcsync0(uint16 slave, boolean act, uint32 CyclTime, uint32 CyclShift)
69 {
70  uint8 h, RA;
71  uint16 wc, slaveh;
72  int64 t, t1;
73  int32 tc;
74 
75  slaveh = ec_slave[slave].configadr;
76  RA = 0;
77 
78  /* stop cyclic operation, ready for next trigger */
79  wc = ec_FPWR(slaveh, ECT_REG_DCSYNCACT, sizeof(RA), &RA, EC_TIMEOUTRET);
80  if (act)
81  {
82  RA = 1 + 2; /* act cyclic operation and sync0, sync1 deactivated */
83  }
84  h = 0;
85  wc = ec_FPWR(slaveh, ECT_REG_DCCUC, sizeof(h), &h, EC_TIMEOUTRET); /* write access to ethercat */
86  wc = ec_FPRD(slaveh, ECT_REG_DCSYSTIME, sizeof(t1), &t1, EC_TIMEOUTRET); /* read local time of slave */
87  t1 = etohll(t1);
88 
89  /* Calculate first trigger time, always a whole multiple of CyclTime rounded up
90  plus the shifttime (can be negative)
91  This insures best sychronisation between slaves, slaves with the same CyclTime
92  will sync at the same moment (you can use CyclShift to shift the sync) */
93  if (CyclTime > 0)
94  {
95  t = ((t1 + SyncDelay) / CyclTime) * CyclTime + CyclTime + CyclShift;
96  }
97  else
98  {
99  t = t1 + SyncDelay + CyclShift;
100  /* first trigger at T1 + CyclTime + SyncDelay + CyclShift in ns */
101  }
102  t = htoell(t);
103  wc = ec_FPWR(slaveh, ECT_REG_DCSTART0, sizeof(t), &t, EC_TIMEOUTRET); /* SYNC0 start time */
104  tc = htoel(CyclTime);
105  wc = ec_FPWR(slaveh, ECT_REG_DCCYCLE0, sizeof(tc), &tc, EC_TIMEOUTRET); /* SYNC0 cycle time */
106  wc = ec_FPWR(slaveh, ECT_REG_DCSYNCACT, sizeof(RA), &RA, EC_TIMEOUTRET); /* activate cyclic operation */
107 }

163 /* latched port time of slave */
164 int32 ec_porttime(uint16 slave, uint8 port)
165 {
166  int32 ts;
167  switch (port)
168  {
169  case 0:
170  ts = ec_slave[slave].DCrtA;
171  break;
172  case 1:
173  ts = ec_slave[slave].DCrtB;
174  break;
175  case 2:
176  ts = ec_slave[slave].DCrtC;
177  break;
178  case 3:
179  ts = ec_slave[slave].DCrtD;
180  break;
181  default:
182  ts = 0;
183  }
184  return ts;
185 }

187 /* calculate previous active port of a slave */
188 uint8 ec_prevport(uint16 slave, uint8 port)
189 {
190  uint8 pport = port;
191  uint8 aport = ec_slave[slave].activeports;
192  switch(port)
193  {
194  case 0:
195  if(aport & PORTM2)
196  pport = 2;
197  else if (aport & PORTM1)
198  pport = 1;
199  else if (aport & PORTM2)
200  pport = 3;
201  break;
202  case 1:
203  if(aport & PORTM3)
204  pport = 3;
205  else if (aport & PORTM0)
206  pport = 0;
207  else if (aport & PORTM2)
208  pport = 2;
209  break;
210  case 2:
211  if(aport & PORTM1)
212  pport = 1;
213  else if (aport & PORTM3)
214  pport = 3;
215  else if (aport & PORTM0)
216  pport = 0;
217  break;
218  case 3:
219  if(aport & PORTM0)
220  pport = 0;
221  else if (aport & PORTM2)
222  pport = 2;
223  else if (aport & PORTM1)
224  pport = 1;
225  break;
226  }
227  return pport;
228 }

230 /* search unconsumed ports in parent, consume and return first open port */
231 uint8 ec_parentport(uint16 parent)
232 {
233  uint8 parentport = 0;
234  uint8 b;
235  /* search order is important, here 3 - 1 - 2 - 0 */
236  b = ec_slave[parent].consumedports;
237  if (b & PORTM3)
238  {
239  parentport = 3;
240  b &= (uint8)~PORTM3;
241  }
242  else if (b & PORTM1)
243  {
244  parentport = 1;
245  b &= (uint8)~PORTM1;
246  }
247  else if (b & PORTM2)
248  {
249  parentport = 2;
250  b &= (uint8)~PORTM2;
251  }
252  else if (b & PORTM0)
253  {
254  parentport = 0;
255  b &= (uint8)~PORTM0;
256  }
257  ec_slave[parent].consumedports = b;
258  return parentport;
259 }

266 boolean ec_configdc(void)
267 {
268  uint16 i, wc, slaveh, parent, child;
269  uint16 parenthold = 0;
270  uint16 prevDCslave = 0;
271  int32 ht, dt1, dt2, dt3;
272  int64 hrt;
273  uint8 entryport;
274  int8 nlist;
275  int8 plist[4];
276  int32 tlist[4];
277 
278  ec_slave[0].hasdc = FALSE;
279  ec_group[0].hasdc = FALSE;
280  ht = 0;
281  ec_BWR(0, ECT_REG_DCTIME0, sizeof(ht), &ht, EC_TIMEOUTRET); /* latch DCrecvTimeA of all slaves */
282  for (i = 1; i <= ec_slavecount; i++)
283  {
284  ec_slave[i].consumedports = ec_slave[i].activeports;
285  if (ec_slave[i].hasdc)
286  {
287  if (!ec_slave[0].hasdc)
288  {
289  ec_slave[0].hasdc = TRUE;
290  ec_slave[0].DCnext = i;
291  ec_slave[i].DCprevious = 0;
292  ec_group[0].hasdc = TRUE;
293  ec_group[0].DCnext = i;
294  }
295  else
296  {
297  ec_slave[prevDCslave].DCnext = i;
298  ec_slave[i].DCprevious = prevDCslave;
299  }
300  /* this branch has DC slave so remove parenthold */
301  parenthold = 0;
302  prevDCslave = i;
303  slaveh = ec_slave[i].configadr;
304  wc = ec_FPRD(slaveh, ECT_REG_DCTIME0, sizeof(ht), &ht, EC_TIMEOUTRET);
305  ec_slave[i].DCrtA = etohl(ht);
306  /* 64bit latched DCrecvTimeA of each specific slave */
307  wc = ec_FPRD(slaveh, ECT_REG_DCSOF, sizeof(hrt), &hrt, EC_TIMEOUTRET);
308  /* use it as offset in order to set local time around 0 */
309  hrt = htoell(-etohll(hrt));
310  /* save it in the offset register */
311  wc = ec_FPWR(slaveh, ECT_REG_DCSYSOFFSET, sizeof(hrt), &hrt, EC_TIMEOUTRET);
312  wc = ec_FPRD(slaveh, ECT_REG_DCTIME1, sizeof(ht), &ht, EC_TIMEOUTRET);
313  ec_slave[i].DCrtB = etohl(ht);
314  wc = ec_FPRD(slaveh, ECT_REG_DCTIME2, sizeof(ht), &ht, EC_TIMEOUTRET);
315  ec_slave[i].DCrtC = etohl(ht);
316  wc = ec_FPRD(slaveh, ECT_REG_DCTIME3, sizeof(ht), &ht, EC_TIMEOUTRET);
317  ec_slave[i].DCrtD = etohl(ht);
318 
319  /* make list of active ports and their time stamps */
320  nlist = 0;
321  if (ec_slave[i].activeports & PORTM0)
322  {
323  plist[nlist] = 0;
324  tlist[nlist] = ec_slave[i].DCrtA;
325  nlist++;
326  }
327  if (ec_slave[i].activeports & PORTM3)
328  {
329  plist[nlist] = 3;
330  tlist[nlist] = ec_slave[i].DCrtD;
331  nlist++;
332  }
333  if (ec_slave[i].activeports & PORTM1)
334  {
335  plist[nlist] = 1;
336  tlist[nlist] = ec_slave[i].DCrtB;
337  nlist++;
338  }
339  if (ec_slave[i].activeports & PORTM2)
340  {
341  plist[nlist] = 2;
342  tlist[nlist] = ec_slave[i].DCrtC;
343  nlist++;
344  }
345  /* entryport is port with the lowest timestamp */
346  entryport = 0;
347  if((nlist > 1) && (tlist[1] < tlist[entryport]))
348  {
349  entryport = 1;
350  }
351  if((nlist > 2) && (tlist[2] < tlist[entryport]))
352  {
353  entryport = 2;
354  }
355  if((nlist > 3) && (tlist[3] < tlist[entryport]))
356  {
357  entryport = 3;
358  }
359  entryport = plist[entryport];
360  ec_slave[i].entryport = entryport;
361  /* consume entryport from activeports */
362  ec_slave[i].consumedports &= (uint8)~(1 << entryport);
363 
364  /* finding DC parent of current */
365  parent = i;
366  do
367  {
368  child = parent;
369  parent = ec_slave[parent].parent;
370  }
371  while (!((parent == 0) || (ec_slave[parent].hasdc)));
372  /* only calculate propagation delay if slave is not the first */
373  if (parent > 0)
374  {
375  /* find port on parent this slave is connected to */
376  ec_slave[i].parentport = ec_parentport(parent);
377  if (ec_slave[parent].topology == 1)
378  {
379  ec_slave[i].parentport = ec_slave[parent].entryport;
380  }
381 
382  dt1 = 0;
383  dt2 = 0;
384  /* delta time of (parentport - 1) - parentport */
385  /* note: order of ports is 0 - 3 - 1 -2 */
386  /* non active ports are skipped */
387  dt3 = ec_porttime(parent, ec_slave[i].parentport) -
388  ec_porttime(parent, ec_prevport(parent, ec_slave[i].parentport));
389  /* current slave has children */
390  /* those childrens delays need to be substacted */
391  if (ec_slave[i].topology > 1)
392  {
393  dt1 = ec_porttime(i, ec_prevport(i, ec_slave[i].entryport)) -
394  ec_porttime(i, ec_slave[i].entryport);
395  }
396  /* we are only interrested in positive diference */
397  if (dt1 > dt3) dt1 = -dt1;
398  /* current slave is not the first child of parent */
399  /* previous childs delays need to be added */
400  if ((child - parent) > 1)
401  {
402  dt2 = ec_porttime(parent, ec_prevport(parent, ec_slave[i].parentport)) -
403  ec_porttime(parent, ec_slave[parent].entryport);
404  }
405  if (dt2 < 0) dt2 = -dt2;
406 
407  /* calculate current slave delay from delta times */
408  /* assumption : forward delay equals return delay */
409  ec_slave[i].pdelay = ((dt3 - dt1) / 2) + dt2 + ec_slave[parent].pdelay;
410  ht = htoel(ec_slave[i].pdelay);
411  /* write propagation delay*/
412  wc = ec_FPWR(slaveh, ECT_REG_DCSYSDELAY, sizeof(ht), &ht, EC_TIMEOUTRET);
413  }
414  }
415  else
416  {
417  ec_slave[i].DCrtA = 0;
418  ec_slave[i].DCrtB = 0;
419  ec_slave[i].DCrtC = 0;
420  ec_slave[i].DCrtD = 0;
421  parent = ec_slave[i].parent;
422  /* if non DC slave found on first position on branch hold root parent */
423  if ( (parent > 0) && (ec_slave[parent].topology > 2))
424  parenthold = parent;
425  /* if branch has no DC slaves consume port on root parent */
426  if ( parenthold && (ec_slave[i].topology == 1))
427  {
428  ec_parentport(parenthold);
429  parenthold = 0;
430  }
431  }
432  }
433 
434  return ec_slave[0].hasdc;
435 }
