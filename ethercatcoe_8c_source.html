179 int ec_SDOread(uint16 slave, uint16 index, uint8 subindex,
180  boolean CA, int *psize, void *p, int timeout)
181 {
182  ec_SDOt *SDOp, *aSDOp;
183  uint16 bytesize, Framedatasize;
184  int wkc;
185  int32 SDOlen;
186  uint8 *bp;
187  uint8 *hp;
188  ec_mbxbuft MbxIn, MbxOut;
189  uint8 cnt, toggle;
190  boolean NotLast;
191 
192  ec_clearmbx(&MbxIn);
193  /* Empty slave out mailbox if something is in. Timout set to 0 */
194  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, 0);
195  ec_clearmbx(&MbxOut);
196  aSDOp = (ec_SDOt *)&MbxIn;
197  SDOp = (ec_SDOt *)&MbxOut;
198  SDOp->MbxHeader.length = htoes(0x000a);
199  SDOp->MbxHeader.address = htoes(0x0000);
200  SDOp->MbxHeader.priority = 0x00;
201  /* get new mailbox count value, used as session handle */
202  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
203  ec_slave[slave].mbx_cnt = cnt;
204  SDOp->MbxHeader.mbxtype = ECT_MBXT_COE + (cnt << 4); /* CoE */
205  SDOp->CANOpen = htoes(0x000 + (ECT_COES_SDOREQ << 12)); /* number 9bits service upper 4 bits (SDO request) */
206  if (CA)
207  SDOp->Command = ECT_SDO_UP_REQ_CA; /* upload request complete access */
208  else
209  SDOp->Command = ECT_SDO_UP_REQ; /* upload request normal */
210  SDOp->Index = htoes(index);
211  if (CA && (subindex > 1))
212  subindex = 1;
213  SDOp->SubIndex = subindex;
214  SDOp->ldata[0] = 0;
215  /* send CoE request to slave */
216  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
217  if (wkc > 0) /* succeeded to place mailbox in slave ? */
218  {
219  /* clean mailboxbuffer */
220  ec_clearmbx(&MbxIn);
221  /* read slave response */
222  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, timeout);
223  if (wkc > 0) /* succeeded to read slave response ? */
224  {
225  /* slave response should be CoE, SDO response and the correct index */
226  if (((aSDOp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_COE) &&
227  ((etohs(aSDOp->CANOpen) >> 12) == ECT_COES_SDORES) &&
228  (aSDOp->Index == SDOp->Index))
229  {
230  if ((aSDOp->Command & 0x02) > 0)
231  {
232  /* expedited frame response */
233  bytesize = 4 - ((aSDOp->Command >> 2) & 0x03);
234  if (*psize >= bytesize) /* parameter buffer big enough ? */
235  {
236  /* copy parameter in parameter buffer */
237  memcpy(p, &aSDOp->ldata[0], bytesize);
238  /* return the real parameter size */
239  *psize = bytesize;
240  }
241  else
242  {
243  wkc = 0;
244  ec_packeterror(slave, index, subindex, 3); /* data container too small for type */
245  }
246  }
247  else
248  { /* normal frame response */
249  SDOlen = etohl(aSDOp->ldata[0]);
250  /* Does parameter fit in parameter buffer ? */
251  if (SDOlen <= *psize)
252  {
253  bp = p;
254  hp = p;
255  /* calculate mailbox transfer size */
256  Framedatasize = (etohs(aSDOp->MbxHeader.length) - 10);
257  if (Framedatasize < SDOlen) /* transfer in segments? */
258  {
259  /* copy parameter data in parameter buffer */
260  memcpy(hp, &aSDOp->ldata[1], Framedatasize);
261  /* increment buffer pointer */
262  hp += Framedatasize;
263  *psize = Framedatasize;
264  NotLast = TRUE;
265  toggle= 0x00;
266  while (NotLast) /* segmented transfer */
267  {
268  SDOp = (ec_SDOt *)&MbxOut;
269  SDOp->MbxHeader.length = htoes(0x000a);
270  SDOp->MbxHeader.address = htoes(0x0000);
271  SDOp->MbxHeader.priority = 0x00;
272  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
273  ec_slave[slave].mbx_cnt = cnt;
274  SDOp->MbxHeader.mbxtype = ECT_MBXT_COE + (cnt << 4); /* CoE */
275  SDOp->CANOpen = htoes(0x000 + (ECT_COES_SDOREQ << 12)); /* number 9bits service upper 4 bits (SDO request) */
276  SDOp->Command = ECT_SDO_SEG_UP_REQ + toggle; /* segment upload request */
277  SDOp->Index = htoes(index);
278  SDOp->SubIndex = subindex;
279  SDOp->ldata[0] = 0;
280  /* send segmented upload request to slave */
281  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
282  /* is mailbox transfered to slave ? */
283  if (wkc > 0)
284  {
285  ec_clearmbx(&MbxIn);
286  /* read slave response */
287  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, timeout);
288  /* has slave responded ? */
289  if (wkc > 0)
290  {
291  /* slave response should be CoE, SDO response */
292  if ((((aSDOp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_COE) &&
293  ((etohs(aSDOp->CANOpen) >> 12) == ECT_COES_SDORES) &&
294  ((aSDOp->Command & 0xe0) == 0x00)))
295  {
296  /* calculate mailbox transfer size */
297  Framedatasize = etohs(aSDOp->MbxHeader.length) - 3;
298  if ((aSDOp->Command & 0x01) > 0)
299  { /* last segment */
300  NotLast = FALSE;
301  if (Framedatasize == 7)
302  /* substract unused bytes from frame */
303  Framedatasize = Framedatasize - ((aSDOp->Command & 0x0e) >> 1);
304  /* copy to parameter buffer */
305  memcpy(hp, &(aSDOp->Index), Framedatasize);
306  }
307  else /* segments follow */
308  {
309  /* copy to parameter buffer */
310  memcpy(hp, &(aSDOp->Index), Framedatasize);
311  /* increment buffer pointer */
312  hp += Framedatasize;
313  }
314  /* update parametersize */
315  *psize += Framedatasize;
316  }
317  /* unexpected frame returned from slave */
318  else
319  {
320  NotLast = FALSE;
321  if ((aSDOp->Command) == ECT_SDO_ABORT) /* SDO abort frame received */
322  ec_SDOerror(slave, index, subindex, etohl(aSDOp->ldata[0]));
323  else
324  ec_packeterror(slave, index, subindex, 1); /* Unexpected frame returned */
325  wkc = 0;
326  }
327  }
328  }
329  toggle = toggle ^ 0x10; /* toggle bit for segment request */
330  }
331  }
332  /* non segmented transfer */
333  else
334  {
335  /* copy to parameter buffer */
336  memcpy(bp, &aSDOp->ldata[1], SDOlen);
337  *psize = SDOlen;
338  }
339  }
340  /* parameter buffer too small */
341  else
342  {
343  wkc = 0;
344  ec_packeterror(slave, index, subindex, 3); /* data container too small for type */
345  }
346  }
347  }
348  /* other slave response */
349  else
350  {
351  if ((aSDOp->Command) == ECT_SDO_ABORT) /* SDO abort frame received */
352  ec_SDOerror(slave, index, subindex, etohl(aSDOp->ldata[0]));
353  else
354  ec_packeterror(slave, index, subindex, 1); /* Unexpected frame returned */
355  wkc = 0;
356  }
357  }
358  }
359  return wkc;
360 }

377 int ec_SDOwrite(uint16 Slave, uint16 Index, uint8 SubIndex,
378  boolean CA, int psize, void *p, int Timeout)
379 {
380  ec_SDOt *SDOp, *aSDOp;
381  int wkc, maxdata;
382  ec_mbxbuft MbxIn, MbxOut;
383  uint8 cnt, toggle;
384  uint16 framedatasize;
385  boolean NotLast;
386  uint8 *hp;
387 
388  ec_clearmbx(&MbxIn);
389  /* Empty slave out mailbox if something is in. Timout set to 0 */
390  wkc = ec_mbxreceive(Slave, (ec_mbxbuft *)&MbxIn, 0);
391  ec_clearmbx(&MbxOut);
392  aSDOp = (ec_SDOt *)&MbxIn;
393  SDOp = (ec_SDOt *)&MbxOut;
394  maxdata = ec_slave[Slave].mbx_l - 0x10; /* data section=mailbox size - 6 mbx - 2 CoE - 8 sdo req */
395  framedatasize = psize;
396  NotLast = FALSE;
397  if (framedatasize > maxdata)
398  {
399  framedatasize = maxdata; /* segmented transfer needed */
400  NotLast = TRUE;
401  }
402  SDOp->MbxHeader.length = htoes(0x0a + framedatasize);
403  SDOp->MbxHeader.address = htoes(0x0000);
404  SDOp->MbxHeader.priority = 0x00;
405  /* get new mailbox counter, used for session handle */
406  cnt = ec_nextmbxcnt(ec_slave[Slave].mbx_cnt);
407  ec_slave[Slave].mbx_cnt = cnt;
408  SDOp->MbxHeader.mbxtype = ECT_MBXT_COE + (cnt << 4); /* CoE */
409  SDOp->CANOpen = htoes(0x000 + (ECT_COES_SDOREQ << 12)); /* number 9bits service upper 4 bits */
410  if (CA)
411  SDOp->Command = ECT_SDO_DOWN_INIT_CA; /* Complete Access, normal SDO init download transfer */
412  else
413  SDOp->Command = ECT_SDO_DOWN_INIT; /* normal SDO init download transfer */
414  SDOp->Index = htoes(Index);
415  SDOp->SubIndex = SubIndex;
416  if (CA && (SubIndex > 1))
417  SDOp->SubIndex = 1;
418  SDOp->ldata[0] = htoel(psize);
419  hp = p;
420  /* copy parameter data to mailbox */
421  memcpy(&SDOp->ldata[1], hp, framedatasize);
422  hp += framedatasize;
423  psize -= framedatasize;
424  /* send mailbox SDO download request to slave */
425  wkc = ec_mbxsend(Slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
426  if (wkc > 0)
427  {
428  ec_clearmbx(&MbxIn);
429  /* read slave response */
430  wkc = ec_mbxreceive(Slave, (ec_mbxbuft *)&MbxIn, Timeout);
431  if (wkc > 0)
432  {
433  /* response should be CoE, SDO response, correct index and subindex */
434  if (((aSDOp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_COE) &&
435  ((etohs(aSDOp->CANOpen) >> 12) == ECT_COES_SDORES) &&
436  (aSDOp->Index == SDOp->Index) &&
437  (aSDOp->SubIndex == SDOp->SubIndex))
438  {
439  /* all ok */
440  maxdata += 7;
441  toggle = 0;
442  /* repeat while segments left */
443  while (NotLast)
444  {
445  SDOp = (ec_SDOt *)&MbxOut;
446  framedatasize = psize;
447  NotLast = FALSE;
448  SDOp->Command = 0x01; /* last segment */
449  if (framedatasize > maxdata)
450  {
451  framedatasize = maxdata; /* more segments needed */
452  NotLast = TRUE;
453  SDOp->Command = 0x00; /* segments follow */
454  }
455  if (!NotLast && (framedatasize < 7))
456  {
457  SDOp->MbxHeader.length = htoes(0x0a); /* minimum size */
458  SDOp->Command = 0x01 + ((7 - framedatasize) << 1); /* last segment reduced octets */
459  }
460  else
461  SDOp->MbxHeader.length = htoes(framedatasize + 3); /* data + 2 CoE + 1 SDO */
462  SDOp->MbxHeader.address = htoes(0x0000);
463  SDOp->MbxHeader.priority = 0x00;
464  /* get new mailbox counter value */
465  cnt = ec_nextmbxcnt(ec_slave[Slave].mbx_cnt);
466  ec_slave[Slave].mbx_cnt = cnt;
467  SDOp->MbxHeader.mbxtype = ECT_MBXT_COE + (cnt << 4); /* CoE */
468  SDOp->CANOpen = htoes(0x000 + (ECT_COES_SDOREQ << 12)); /* number 9bits service upper 4 bits (SDO request) */
469  SDOp->Command = SDOp->Command + toggle; /* add toggle bit to command byte */
470  /* copy parameter data to mailbox */
471  memcpy(&SDOp->Index, hp, framedatasize);
472  /* update parameter buffer pointer */
473  hp += framedatasize;
474  psize -= framedatasize;
475  /* send SDO download request */
476  wkc = ec_mbxsend(Slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
477  if (wkc > 0)
478  {
479  ec_clearmbx(&MbxIn);
480  /* read slave response */
481  wkc = ec_mbxreceive(Slave, (ec_mbxbuft *)&MbxIn, Timeout);
482  if (wkc > 0)
483  {
484  if (((aSDOp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_COE) &&
485  ((etohs(aSDOp->CANOpen) >> 12) == ECT_COES_SDORES) &&
486  ((aSDOp->Command & 0xe0) == 0x20))
487  {
488  /* all OK, nothing to do */
489  }
490  else
491  {
492  if (aSDOp->Command == ECT_SDO_ABORT) /* SDO abort frame received */
493  ec_SDOerror(Slave, Index, SubIndex, etohl(aSDOp->ldata[0]));
494  else
495  ec_packeterror(Slave, Index, SubIndex, 1); /* Unexpected frame returned */
496  wkc = 0;
497  NotLast = FALSE;
498  }
499  }
500  }
501  toggle = toggle ^ 0x10; /* toggle bit for segment request */
502  }
503  }
504  /* unexpected response from slave */
505  else
506  {
507  if (aSDOp->Command == ECT_SDO_ABORT) /* SDO abort frame received */
508  ec_SDOerror(Slave, Index, SubIndex, etohl(aSDOp->ldata[0]));
509  else
510  ec_packeterror(Slave, Index, SubIndex, 1); /* Unexpected frame returned */
511  wkc = 0;
512  }
513  }
514  }
515 
516  return wkc;
517 }

964  n = (etohs(aSDOp->MbxHeader.length) - (6 + 2)) / 2;

985  stop = FALSE;

987  offset = 0;
988  }

995  stop = TRUE;
