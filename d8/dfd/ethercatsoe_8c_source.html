111 int ec_SoEread(uint16 slave, uint8 driveNo, uint8 elementflags, uint16 idn, int *psize, void *p, int timeout)
112 {
113  ec_SoEt *SoEp, *aSoEp;
114  uint16 totalsize, framedatasize;
115  int wkc;
116  uint8 *bp;
117  uint8 *mp;
118  uint16 *errorcode;
119  ec_mbxbuft MbxIn, MbxOut;
120  uint8 cnt;
121  boolean NotLast;
122 
123  ec_clearmbx(&MbxIn);
124  /* Empty slave out mailbox if something is in. Timeout set to 0 */
125  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, 0);
126  ec_clearmbx(&MbxOut);
127  aSoEp = (ec_SoEt *)&MbxIn;
128  SoEp = (ec_SoEt *)&MbxOut;
129  SoEp->MbxHeader.length = htoes(sizeof(ec_SoEt) - sizeof(ec_mbxheadert));
130  SoEp->MbxHeader.address = htoes(0x0000);
131  SoEp->MbxHeader.priority = 0x00;
132  /* get new mailbox count value, used as session handle */
133  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
134  ec_slave[slave].mbx_cnt = cnt;
135  SoEp->MbxHeader.mbxtype = ECT_MBXT_SOE + (cnt << 4); /* SoE */
136  SoEp->opCode = ECT_SOE_READREQ;
137  SoEp->incomplete = 0;
138  SoEp->error = 0;
139  SoEp->driveNo = driveNo;
140  SoEp->elementflags = elementflags;
141  SoEp->idn = htoes(idn);
142  totalsize = 0;
143  bp = p;
144  mp = (uint8 *)&MbxIn + sizeof(ec_SoEt);
145  NotLast = TRUE;
146  /* send SoE request to slave */
147  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
148  if (wkc > 0) /* succeeded to place mailbox in slave ? */
149  {
150  while (NotLast)
151  {
152  /* clean mailboxbuffer */
153  ec_clearmbx(&MbxIn);
154  /* read slave response */
155  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, timeout);
156  if (wkc > 0) /* succeeded to read slave response ? */
157  {
158  /* slave response should be SoE, ReadRes */
159  if (((aSoEp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_SOE) &&
160  (aSoEp->opCode == ECT_SOE_READRES) &&
161  (aSoEp->error == 0) &&
162  (aSoEp->driveNo == driveNo) &&
163  (aSoEp->elementflags == elementflags))
164  {
165  framedatasize = etohs(aSoEp->MbxHeader.length) - sizeof(ec_SoEt) + sizeof(ec_mbxheadert);
166  totalsize += framedatasize;
167  /* Does parameter fit in parameter buffer ? */
168  if (totalsize <= *psize)
169  {
170  /* copy parameter data in parameter buffer */
171  memcpy(bp, mp, framedatasize);
172  /* increment buffer pointer */
173  bp += framedatasize;
174  }
175  else
176  {
177  framedatasize -= totalsize - *psize;
178  totalsize = *psize;
179  /* copy parameter data in parameter buffer */
180  if (framedatasize > 0) memcpy(bp, mp, framedatasize);
181  }
182 
183  if (!aSoEp->incomplete)
184  {
185  NotLast = FALSE;
186  *psize = totalsize;
187  }
188  }
189  /* other slave response */
190  else
191  {
192  NotLast = FALSE;
193  if (((aSoEp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_SOE) &&
194  (aSoEp->opCode == ECT_SOE_READRES) &&
195  (aSoEp->error == 1))
196  {
197  mp = (uint8 *)&MbxIn + (etohs(aSoEp->MbxHeader.length) + sizeof(ec_mbxheadert) - sizeof(uint16));
198  errorcode = (uint16 *)mp;
199  ec_SoEerror(slave, idn, *errorcode);
200  }
201  else
202  {
203  ec_packeterror(slave, idn, 0, 1); /* Unexpected frame returned */
204  }
205  wkc = 0;
206  }
207  }
208  else
209  {
210  NotLast = FALSE;
211  ec_packeterror(slave, idn, 0, 4); /* no response */
212  }
213  }
214  }
215  return wkc;
216 }

235  uint16 framedatasize, maxdata;
236  int wkc;
237  uint8 *mp;
238  uint8 *hp;
239  uint16 *errorcode;

241  uint8 cnt;
242  boolean NotLast;
243 
244  ec_clearmbx(&MbxIn);
245  /* Empty slave out mailbox if something is in. Timeout set to 0 */
246  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, 0);
247  ec_clearmbx(&MbxOut);
248  aSoEp = (ec_SoEt *)&MbxIn;
249  SoEp = (ec_SoEt *)&MbxOut;
250  SoEp->MbxHeader.address = htoes(0x0000);
251  SoEp->MbxHeader.priority = 0x00;
252  SoEp->opCode = ECT_SOE_WRITEREQ;
253  SoEp->error = 0;
254  SoEp->driveNo = driveNo;
255  SoEp->elementflags = elementflags;
256  hp = p;
257  mp = (uint8 *)&MbxOut + sizeof(ec_SoEt);
258  maxdata = ec_slave[slave].mbx_l - sizeof(ec_SoEt);
259  NotLast = TRUE;
260  while (NotLast)
261  {
262  framedatasize = psize;
263  NotLast = FALSE;
264  SoEp->idn = htoes(idn);
265  SoEp->incomplete = 0;
266  if (framedatasize > maxdata)
267  {
268  framedatasize = maxdata; /* segmented transfer needed */
269  NotLast = TRUE;
270  SoEp->incomplete = 1;
271  SoEp->fragmentsleft = psize / maxdata;
272  }
273  SoEp->MbxHeader.length = htoes(sizeof(ec_SoEt) - sizeof(ec_mbxheadert) + framedatasize);
274  /* get new mailbox counter, used for session handle */
275  cnt = ec_nextmbxcnt(ec_slave[slave].mbx_cnt);
276  ec_slave[slave].mbx_cnt = cnt;
277  SoEp->MbxHeader.mbxtype = ECT_MBXT_SOE + (cnt << 4); /* SoE */
278  /* copy parameter data to mailbox */
279  memcpy(mp, hp, framedatasize);
280  hp += framedatasize;
281  psize -= framedatasize;
282  /* send SoE request to slave */
283  wkc = ec_mbxsend(slave, (ec_mbxbuft *)&MbxOut, EC_TIMEOUTTXM);
284  if (wkc > 0) /* succeeded to place mailbox in slave ? */
285  {
286  if (!NotLast || !ec_mbxempty(slave, timeout))
287  {
288  /* clean mailboxbuffer */
289  ec_clearmbx(&MbxIn);
290  /* read slave response */
291  wkc = ec_mbxreceive(slave, (ec_mbxbuft *)&MbxIn, timeout);
292  if (wkc > 0) /* succeeded to read slave response ? */
293  {
294  NotLast = FALSE;
295  /* slave response should be SoE, WriteRes */
296  if (((aSoEp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_SOE) &&
297  (aSoEp->opCode == ECT_SOE_WRITERES) &&
298  (aSoEp->error == 0) &&
299  (aSoEp->driveNo == driveNo) &&
300  (aSoEp->elementflags == elementflags))
301  {
302  /* SoE write succeeded */
303  }
304  /* other slave response */
305  else
306  {
307  if (((aSoEp->MbxHeader.mbxtype & 0x0f) == ECT_MBXT_SOE) &&
308  (aSoEp->opCode == ECT_SOE_READRES) &&
309  (aSoEp->error == 1))
310  {
311  mp = (uint8 *)&MbxIn + (etohs(aSoEp->MbxHeader.length) + sizeof(ec_mbxheadert) - sizeof(uint16));
312  errorcode = (uint16 *)mp;
313  ec_SoEerror(slave, idn, *errorcode);
314  }
315  else
316  {
317  ec_packeterror(slave, idn, 0, 1); /* Unexpected frame returned */
318  }
319  wkc = 0;
320  }
321  }
322  else
323  {
324  ec_packeterror(slave, idn, 0, 4); /* no response */
325  }
326  }
327  }
328  }
329  return wkc;
330 }

343 int ec_readIDNmap(uint16 slave, int *Osize, int *Isize)
344 {
345  int retVal = 0;
346  int wkc;
347  int psize;
348  uint16 entries, itemcount;
349 
350  *Isize = 0;
351  *Osize = 0;
352  psize = sizeof(SoEmapping);
353  /* read output mapping via SoE */
354  wkc = ec_SoEread(slave, 0, EC_SOE_VALUE_B, EC_IDN_MDTCONFIG, &psize, &SoEmapping, EC_TIMEOUTRXM);
355  if ((wkc > 0) && (psize >= 4) && ((entries = etohs(SoEmapping.currentlength) / 2) > 0) && (entries <= EC_SOE_MAXMAPPING))
356  {
357  /* command word (uint16) is always mapped but not in list */
358  *Osize = 16;
359  for (itemcount = 0 ; itemcount < entries ; itemcount++)
360  {
361  psize = sizeof(SoEattribute);
362  /* read attribute of each IDN in mapping list */
363  wkc = ec_SoEread(slave, 0, EC_SOE_ATTRIBUTE_B, SoEmapping.idn[itemcount], &psize, &SoEattribute, EC_TIMEOUTRXM);
364  if ((wkc > 0) && (!SoEattribute.list))
365  {
366  /* length : 0 = 8bit, 1 = 16bit .... */
367  *Osize += (int)8 << SoEattribute.length;
368  }
369  }
370  }
371  psize = sizeof(SoEmapping);
372  /* read input mapping via SoE */
373  wkc = ec_SoEread(slave, 0, EC_SOE_VALUE_B, EC_IDN_ATCONFIG, &psize, &SoEmapping, EC_TIMEOUTRXM);
374  if ((wkc > 0) && (psize >= 4) && ((entries = etohs(SoEmapping.currentlength) / 2) > 0) && (entries <= EC_SOE_MAXMAPPING))
375  {
376  /* status word (uint16) is always mapped but not in list */
377  *Isize = 16;
378  for (itemcount = 0 ; itemcount < entries ; itemcount++)
379  {
380  psize = sizeof(SoEattribute);
381  /* read attribute of each IDN in mapping list */
382  wkc = ec_SoEread(slave, 0, EC_SOE_ATTRIBUTE_B, SoEmapping.idn[itemcount], &psize, &SoEattribute, EC_TIMEOUTRXM);
383  if ((wkc > 0) && (!SoEattribute.list))
384  {
385  /* length : 0 = 8bit, 1 = 16bit .... */
386  *Isize += (int)8 << SoEattribute.length;
387  }
388  }
389  }
390 
391  /* found some I/O bits ? */
392  if ((*Isize > 0) || (*Osize > 0))
393  {
394  retVal = 1;
395  }
396  return retVal;
397 }
