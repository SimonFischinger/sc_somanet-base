271 char* ec_elist2string(void)
272 {
273  ec_errort Ec;
274  char timestr[20];
275 
276  if (ec_poperror(&Ec))
277  {
278  sprintf(timestr, "Time:%12.3f", Ec.Time.tv_sec + (Ec.Time.tv_usec / 1000000.0) );
279  switch (Ec.Etype)
280  {
281  case EC_ERR_TYPE_SDO_ERROR:
282  sprintf(estring, "%s SDO slave:%d index:%4.4x.%2.2x error:%8.8x %s\n",
283  timestr, Ec.Slave, Ec.Index, Ec.SubIdx, Ec.AbortCode, ec_sdoerror2string(Ec.AbortCode));
284  break;
285  case EC_ERR_TYPE_EMERGENCY:
286  sprintf(estring, "%s EMERGENCY slave:%d error:%4.4x\n",
287  timestr, Ec.Slave, Ec.ErrorCode);
288  break;
289  case EC_ERR_TYPE_PACKET_ERROR:
290  sprintf(estring, "%s PACKET slave:%d index:%4.4x.%2.2x error:%d\n",
291  timestr, Ec.Slave, Ec.Index, Ec.SubIdx, Ec.ErrorCode);
292  break;
293  case EC_ERR_TYPE_SDOINFO_ERROR:
294  sprintf(estring, "%s SDO slave:%d index:%4.4x.%2.2x error:%8.8x %s\n",
295  timestr, Ec.Slave, Ec.Index, Ec.SubIdx, Ec.AbortCode, ec_sdoerror2string(Ec.AbortCode));
296  break;
297  case EC_ERR_TYPE_SOE_ERROR:
298  sprintf(estring, "%s SoE slave:%d IDN:%4.4x error:%4.4x %s\n",
299  timestr, Ec.Slave, Ec.Index, Ec.AbortCode, ec_soeerror2string(Ec.ErrorCode));
300  break;
301  default:
302  sprintf(estring, "%s error:%8.8x\n",
303  timestr, Ec.AbortCode);
304  }
305  return (char*) estring;
306  }
307  else return "";
308 }
