271 uint8 ec_siigetbyte(uint16 slave, uint16 address)
272 {
273  uint16 configadr, eadr;
274  uint64 edat;
275  uint16 mapw, mapb;
276  int lp,cnt;
277  uint8 retval;
278 
279  retval = 0xff;
280  if (slave != esislave) /* not the same slave? */
281  {
282  memset(esimap,0x00,EC_MAXEEPBITMAP); /* clear esibuf cache map */
283  esislave=slave;
284  }
285  if (address < EC_MAXEEPBUF)
286  {
287  mapw = address >> 5;
288  mapb = address - (mapw << 5);
289  if (esimap[mapw] & (uint32)(1 << mapb))
290  {
291  /* byte is already in buffer */
292  retval = esibuf[address];
293  }
294  else
295  {
296  /* byte is not in buffer, put it there */
297  configadr = ec_slave[slave].configadr;
298  ec_eeprom2master(slave); /* set eeprom control to master */
299  eadr = address >> 1;
300  edat = ec_readeepromFP (configadr, eadr, EC_TIMEOUTEEP);
301  /* 8 byte response */
302  if (ec_slave[slave].eep_8byte)
303  {
304  put_unaligned64(edat, &esibuf[eadr << 1]);
305  cnt = 8;
306  }
307  /* 4 byte response */
308  else
309  {
310  put_unaligned32(edat, &esibuf[eadr << 1]);
311  cnt = 4;
312  }
313  /* find bitmap location */
314  mapw = eadr >> 4;
315  mapb = (eadr << 1) - (mapw << 5);
316  for(lp = 0 ; lp < cnt ; lp++)
317  {
318  /* set bitmap for each byte that is read */
319  esimap[mapw] |= (1 << mapb);
320  mapb++;
321  if (mapb > 31)
322  {
323  mapb = 0;
324  mapw++;
325  }
326  }
327  retval = esibuf[address];
328  }
329  }
330 
331  return retval;
332 }
