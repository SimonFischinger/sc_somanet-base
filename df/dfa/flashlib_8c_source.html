563 /* Returns information about the first boot image (the factory image). */
564 int fl_getFactoryImage( fl_BootImageInfo* bootImageInfo )
565 {
566  unsigned tmpBuf[9];
567  fl_int_readBytes(g_flashAccess, 0, (unsigned char*)tmpBuf, 4);
568  unsigned startAddr = (sortBits(tmpBuf[0])+2)<<2; /* Normal case. */
569  fl_int_readBytes(g_flashAccess, startAddr, (unsigned char*)tmpBuf, (6 + 3) * sizeof(int));
570  unsigned *header = tmpBuf;
571  if (sortBits(tmpBuf[0]) != IMAGETAG) {
572  header += 3; /* Secure case. */
573  if (sortBits(header[0]) != IMAGETAG) {
574  return 1;
575  }
576  }
577  bootImageInfo->startAddress = startAddr;
578  bootImageInfo->size = sortBits(header[5]); /* Size is to next sector start. */
579  bootImageInfo->version = sortBits(header[4]);
580  bootImageInfo->factory = 1;
581  return 0;

668 static int
669 fl_initImageWriteState(unsigned address, unsigned maxsize,
670  fl_BootImageInfo* bootImageInfo)
671 {
672  /* Initialise state. */
673  unsigned baseSectorNum = getSectorAtOrAfter(address);
674  /* Don't allow images starting at sector 0 as 0 is overloaded to mean we are
675  * not in the middle of writing an image. We shouldn't be erasing sector 0
676  * anyway since this will erase the second stage loader. */
677  if (baseSectorNum <= 0) {
678  return -1;
679  }
680  fl_imageWriteState.baseAddress = fl_getSectorAddress(baseSectorNum);
681  unsigned pageSize = fl_getPageSize();
682  unsigned limitAddress = fl_imageWriteState.baseAddress + maxsize;
683  limitAddress = ((limitAddress + pageSize - 1) / pageSize) * pageSize;
684  fl_imageWriteState.limitAddress = limitAddress;
685  fl_imageWriteState.currentAddress = fl_imageWriteState.baseAddress;
686  /* Check we won't overwrite anything. Bootable images and the store partition
687  * start on sector boundaries so there is no need to round up the limit
688  * address to the end of sector for the comparison. */
689  unsigned nextAddress;
690  fl_BootImageInfo nextImage = *bootImageInfo;
691  if (fl_getNextBootImage(&nextImage) == 0) {
692  nextAddress = nextImage.startAddress;
693  } else {
694  nextAddress = fl_getDataPartitionBase();
695  }
696  if (fl_imageWriteState.limitAddress > nextAddress) {
697  return -1;
698  }
699  fl_imageWriteState.eraseSector = baseSectorNum;
700  return 1;
701 }

951  return 1;
