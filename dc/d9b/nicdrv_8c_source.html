188 int ec_setupnic(const char * ifname, int secondary)
189 {
190  int i;
191  int r, rval, ifindex, fl;
192  struct timeval timeout;
193  struct ifreq ifr;
194  struct sockaddr_ll sll;
195  int *psock;
196 
197  rval = 0;
198  if (secondary)
199  {
200  /* when using secondary socket it is automatically a redundant setup */
201  psock = &sockhandle2;
202  ec_redstate = ECT_RED_DOUBLE;
203  }
204  else
205  {
206  psock = &sockhandle;
207  ec_redstate = ECT_RED_NONE;
208  }
209  /* we use RAW packet socket, with packet type ETH_P_ECAT */
210  *psock = socket(PF_PACKET, SOCK_RAW, htons(ETH_P_ECAT));
211  timeout.tv_sec = 0;
212  timeout.tv_usec = 1;
213 
214  r = setsockopt(*psock, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout));
215  r = setsockopt(*psock, SOL_SOCKET, SO_SNDTIMEO, &timeout, sizeof(timeout));
216  i = 1;
217  r = setsockopt(*psock, SOL_SOCKET, SO_DONTROUTE, &i, sizeof(i));
218  /* connect socket to NIC by name */
219  strcpy(ifr.ifr_name, ifname);
220  r = ioctl(*psock, SIOCGIFINDEX, &ifr);
221  ifindex = ifr.ifr_ifindex;
222  strcpy(ifr.ifr_name, ifname);
223  ifr.ifr_flags = 0;
224  /* reset flags of NIC interface */
225  r = ioctl(*psock, SIOCGIFFLAGS, &ifr);
226  /* set flags of NIC interface, here promiscuous and broadcast */
227  ifr.ifr_flags = ifr.ifr_flags || IFF_PROMISC || IFF_BROADCAST;
228  r = ioctl(*psock, SIOCGIFFLAGS, &ifr);
229  /* bind socket to protocol, in this case RAW EtherCAT */
230  sll.sll_family = AF_PACKET;
231  sll.sll_ifindex = ifindex;
232  sll.sll_protocol = htons(ETH_P_ECAT);
233  r = bind(*psock, (struct sockaddr *)&sll, sizeof(sll));
234  /* get flags */
235  fl = fcntl(*psock, F_GETFL, 0);
236  /* set nodelay option, so make socket non-blocking */
237 // r = fcntl(*psock, F_SETFL, fl | O_NDELAY);
238  /* setup ethernet headers in tx buffers so we don't have to repeat it */
239  for (i = 0; i < EC_MAXBUF; i++)
240  {
241  ec_setupheader(&ec_txbuf[i]);
242  ec_rxbufstat[i] = EC_BUF_EMPTY;
243  }
244  ec_setupheader(&ec_txbuf2);
245  ec_errcnt = ec_incnt = 0;
246  if (r == 0) rval = 1;
247 
248  return rval;
249 }

410 int ec_inframe(uint8 idx, int stacknumber)
411 {
412  uint16 l;
413  int rval;
414  uint8 idxf;
415  ec_etherheadert *ehp;
416  ec_comt *ecp;
417  ec_stackT *stack;
418  ec_bufT *rxbuf;
419 
420  stack = &ec_stack[stacknumber];
421  rval = EC_NOFRAME;
422  rxbuf = &(*stack->rxbuf)[idx];
423  /* check if requested index is already in buffer ? */
424  if ((idx < EC_MAXBUF) && ( (*stack->rxbufstat)[idx] == EC_BUF_RCVD))
425  {
426  l = (*rxbuf)[0] + ((uint16)((*rxbuf)[1] & 0x0f) << 8);
427  /* return WKC */
428  rval = ((*rxbuf)[l] + ((uint16)(*rxbuf)[l + 1] << 8));
429  /* mark as completed */
430  (*stack->rxbufstat)[idx] = EC_BUF_COMPLETE;
431  }
432  else
433  {
434  pthread_mutex_lock( &ec_rx_mutex );
435  /* non blocking call to retrieve frame from socket */
436  if (ec_recvpkt( stacknumber))
437  {
438  rval = EC_OTHERFRAME;
439  ehp =(ec_etherheadert*)(stack->tempbuf);
440  /* check if it is an EtherCAT frame */
441  if (ehp->etype == htons(ETH_P_ECAT))
442  {
443  ec_incnt++;
444  ecp =(ec_comt*)(&(*stack->tempbuf)[ETH_HEADERSIZE]);
445  l = etohs(ecp->elength) & 0x0fff;
446  idxf = ecp->index;
447  /* found index equals reqested index ? */
448  if (idxf == idx)
449  {
450  /* yes, put it in the buffer array (strip ethernet header) */
451  memcpy(rxbuf, &(*stack->tempbuf)[ETH_HEADERSIZE], (*stack->txbuflength)[idx] - ETH_HEADERSIZE);
452  /* return WKC */
453  rval = ((*rxbuf)[l] + ((uint16)((*rxbuf)[l + 1]) << 8));
454  /* mark as completed */
455  (*stack->rxbufstat)[idx] = EC_BUF_COMPLETE;
456  /* store MAC source word 1 for redundant routing info */
457  (*stack->rxsa)[idx] = ntohs(ehp->sa1);
458  }
459  else
460  {
461  /* check if index exist? */
462  if (idxf < EC_MAXBUF)
463  {
464  rxbuf = &(*stack->rxbuf)[idxf];
465  /* put it in the buffer array (strip ethernet header) */
466  memcpy(rxbuf, &(*stack->tempbuf)[ETH_HEADERSIZE], (*stack->txbuflength)[idxf] - ETH_HEADERSIZE);
467  /* mark as received */
468  (*stack->rxbufstat)[idxf] = EC_BUF_RCVD;
469  (*stack->rxsa)[idxf] = ntohs(ehp->sa1);
470  }
471  else
472  {
473  /* strange things happend */
474  ec_errcnt++;
475  }
476  }
477  }
478  }
479  pthread_mutex_unlock( &ec_rx_mutex );
480 
481  }
482 
483  /* WKC if mathing frame found */
484  return rval;
485 }

498 static int ec_waitinframe_red(uint8 idx, struct timeval tvs)
499 {
500  struct timeval tv1,tv2;
501  int wkc = EC_NOFRAME;
502  int wkc2 = EC_NOFRAME;
503  int primrx, secrx;
504 
505  /* if not in redundat mode then always assume secondary is OK */
506  if (ec_redstate == ECT_RED_NONE)
507  wkc2 = 0;
508  do
509  {
510  /* only read frame if not already in */
511  if (wkc <= EC_NOFRAME)
512  wkc = ec_inframe(idx, 0);
513  /* only try secondary if in redundant mode */
514  if (ec_redstate != ECT_RED_NONE)
515  {
516  /* only read frame if not already in */
517  if (wkc2 <= EC_NOFRAME)
518  wkc2 = ec_inframe(idx, 1);
519  }
520  gettimeofday(&tv1, 0);
521  /* wait for both frames to arrive or timeout */
522  } while (((wkc <= EC_NOFRAME) || (wkc2 <= EC_NOFRAME)) && (timercmp(&tv1, &tvs, <)));
523  /* only do redundant functions when in redundant mode */
524  if (ec_redstate != ECT_RED_NONE)
525  {
526  /* primrx if the reveived MAC source on primary socket */
527  primrx = 0;
528  if (wkc > EC_NOFRAME) primrx = ec_rxsa[idx];
529  /* secrx if the reveived MAC source on psecondary socket */
530  secrx = 0;
531  if (wkc2 > EC_NOFRAME) secrx = ec_rxsa2[idx];
532 
533  /* primary socket got secondary frame and secondary socket got primary frame */
534  /* normal situation in redundant mode */
535  if ( ((primrx == RX_SEC) && (secrx == RX_PRIM)) )
536  {
537  /* copy secondary buffer to primary */
538  memcpy(&ec_rxbuf[idx], &ec_rxbuf2[idx], ec_txbuflength[idx] - ETH_HEADERSIZE);
539  wkc = wkc2;
540  }
541  /* primary socket got nothing or primary frame, and secondary socket got secondary frame */
542  /* we need to resend TX packet */
543  if ( ((primrx == 0) && (secrx == RX_SEC)) ||
544  ((primrx == RX_PRIM) && (secrx == RX_SEC)) )
545  {
546  /* If both primary and secondary have partial connection retransmit the primary received
547  * frame over the secondary socket. The result from the secondary received frame is a combined
548  * frame that traversed all slaves in standard order. */
549  if ( (primrx == RX_PRIM) && (secrx == RX_SEC) )
550  {
551  /* copy primary rx to tx buffer */
552  memcpy(&ec_txbuf[idx][ETH_HEADERSIZE], &ec_rxbuf[idx], ec_txbuflength[idx] - ETH_HEADERSIZE);
553  }
554  gettimeofday(&tv1, 0);
555  tv2.tv_sec = 0;
556  tv2.tv_usec = EC_TIMEOUTRET;
557  timeradd(&tv1, &tv2, &tvs);
558  /* resend secondary tx */
559  ec_outframe(idx,1);
560  do
561  {
562  /* retrieve frame */
563  wkc2 = ec_inframe(idx, 1);
564  gettimeofday(&tv1, 0);
565  } while ((wkc2 <= EC_NOFRAME) && (timercmp(&tv1, &tvs, <)));
566  if (wkc2 > EC_NOFRAME)
567  {
568  /* copy secondary result to primary rx buffer */
569  memcpy(&ec_rxbuf[idx], &ec_rxbuf2[idx], ec_txbuflength[idx] - ETH_HEADERSIZE);
570  wkc = wkc2;
571  }
572  }
573  }
574 
575  /* return WKC or EC_NOFRAME */
576  return wkc;
577 }

615 int ec_srconfirm(uint8 idx, int timeout)
616 {
617  int wkc = EC_NOFRAME;
618  struct timeval tv1, tv2, tv3, tve, tvs, tvh;
619 
620  gettimeofday(&tv1, 0);
621  tv2.tv_sec = 0;
622  tv2.tv_usec = timeout;
623  timeradd(&tv1, &tv2, &tve);
624  do

626  /* tx frame on primary and if in redundant mode a dummy on secondary */
627  ec_outframe_red(idx);
628  gettimeofday(&tv2, 0);
629  timersub(&tv2, &tv1, &tvh);
630  hlp_txtime += (int)tvh.tv_usec;

634  tv1.tv_usec = timeout;

638  /* normally use partial timout for rx */
639  tv1.tv_usec = EC_TIMEOUTRET;

642  /* get frame from primary or if in redundant mode possibly from secondary */
