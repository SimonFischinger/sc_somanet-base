69 int ec_setupdatagram(void *frame, uint8 com, uint8 idx, uint16 ADP, uint16 ADO, uint16 length, void *data)
70 {
71  ec_comt *datagramP;
72  uint8 *frameP;
73 
74  frameP = frame;
75  /* Ethernet header is preset and fixed in frame buffers
76  EtherCAT header needs to be added after that */
77  datagramP = (ec_comt*)&frameP[ETH_HEADERSIZE];
78  datagramP->elength = htoes(EC_ECATTYPE + EC_HEADERSIZE + length);
79  datagramP->command = com;
80  datagramP->index = idx;
81  datagramP->ADP = htoes(ADP);
82  datagramP->ADO = htoes(ADO);
83  datagramP->dlength = htoes(length);
84  if (length > 0)
85  {
86  memcpy(&frameP[ETH_HEADERSIZE + EC_HEADERSIZE], data, length);
87  }
88  /* set WKC to zero */
89  frameP[ETH_HEADERSIZE + EC_HEADERSIZE + length] = 0x00;
90  frameP[ETH_HEADERSIZE + EC_HEADERSIZE + length + 1] = 0x00;
91  /* set size of frame in buffer array */
92  ec_txbuflength[idx] = ETH_HEADERSIZE + EC_HEADERSIZE + EC_WKCSIZE + length;
93 
94  return 0;
95 }

109 int ec_adddatagram(void *frame, uint8 com, uint8 idx, boolean more, uint16 ADP, uint16 ADO, uint16 length, void *data)
110 {
111  ec_comt *datagramP;
112  uint8 *frameP;
113  uint16 prevlength;
114 
115  frameP = frame;
116  /* copy previous frame size */
117  prevlength = ec_txbuflength[idx];
118  datagramP = (ec_comt*)&frameP[ETH_HEADERSIZE];
119  /* add new datagram to ethernet frame size */
120  datagramP->elength = htoes( etohs(datagramP->elength) + EC_HEADERSIZE + length );
121  /* add "datagram follows" flag to previous subframe dlength */
122  datagramP->dlength = htoes( etohs(datagramP->dlength) | EC_DATAGRAMFOLLOWS );
123  /* set new EtherCAT header position */
124  datagramP = (ec_comt*)&frameP[prevlength - EC_ELENGTHSIZE];
125  datagramP->command = com;
126  datagramP->index = idx;
127  datagramP->ADP = htoes(ADP);
128  datagramP->ADO = htoes(ADO);
129  if (more)
130  {
131  /* this is not the last datagram to add */
132  datagramP->dlength = htoes(length | EC_DATAGRAMFOLLOWS);
133  }
134  else
135  {
136  /* this is the last datagram in the frame */
137  datagramP->dlength = htoes(length);
138  }
139  if (length > 0)
140  {
141  memcpy(&frameP[prevlength + EC_HEADERSIZE - EC_ELENGTHSIZE], data, length);
142  }
143  /* set WKC to zero */
144  frameP[prevlength + EC_HEADERSIZE - EC_ELENGTHSIZE + length] = 0x00;
145  frameP[prevlength + EC_HEADERSIZE - EC_ELENGTHSIZE + length + 1] = 0x00;
146  /* set size of frame in buffer array */
147  ec_txbuflength[idx] = prevlength + EC_HEADERSIZE - EC_ELENGTHSIZE + EC_WKCSIZE + length;
148 
149  /* return offset to data in rx frame
150  14 bytes smaller than tx frame due to stripping of ethernet header */
151  return prevlength + EC_HEADERSIZE - EC_ELENGTHSIZE - ETH_HEADERSIZE;
152 }

163 int ec_BWR(uint16 ADP, uint16 ADO, uint16 length, void *data, int timeout)
164 {

502 int ec_LRWDC(uint32 LogAdr, uint16 length, void *data, uint16 DCrs, int64 *DCtime, int timeout)
503 {
504  uint16 DCtO;
505  uint8 idx;
506  int wkc;
507  uint64 DCtE;
508 
509  idx = ec_getindex();
510  /* LRW in first datagram */
511  ec_setupdatagram(&ec_txbuf[idx], EC_CMD_LRW, idx, LO_WORD(LogAdr), HI_WORD(LogAdr), length, data);
512  /* FPRMW in second datagram */
513  DCtE = htoell(*DCtime);
514  DCtO = ec_adddatagram(&ec_txbuf[idx], EC_CMD_FRMW, idx, FALSE, DCrs, ECT_REG_DCSYSTIME, sizeof(DCtime), &DCtE);
515  wkc = ec_srconfirm(idx, timeout);
516  if ((wkc > 0) && (ec_rxbuf[idx][EC_CMDOFFSET] == EC_CMD_LRW))
517  {
518  memcpy(data, &ec_rxbuf[idx][EC_HEADERSIZE], length);
519  memcpy(&wkc, &ec_rxbuf[idx][EC_HEADERSIZE + length], EC_WKCSIZE);
520  memcpy(&DCtE, &ec_rxbuf[idx][DCtO], sizeof(*DCtime));
521  *DCtime = etohll(DCtE);
522  }
523  ec_setbufstat(idx, EC_BUF_EMPTY);
524 
525  return wkc;
526 }
